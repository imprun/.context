import Editor from '@monaco-editor/react';

interface JsonEditorProps {
  value: Record<string, any>;
  onChange: (value: Record<string, any>) => void;
  schema?: z.ZodSchema;
  height?: string;
}

export function JsonEditor({ value, onChange, schema, height = '200px' }: JsonEditorProps) {
  const [jsonString, setJsonString] = useState(JSON.stringify(value, null, 2));
  const [error, setError] = useState<string | null>(null);

  const handleChange = (newValue: string | undefined) => {
    if (!newValue) return;
    setJsonString(newValue);

    try {
      const parsed = JSON.parse(newValue);
      if (schema) {
        const result = schema.safeParse(parsed);
        if (!result.success) {
          setError(result.error.errors[0].message);
          return;
        }
      }
      setError(null);
      onChange(parsed);
    } catch (e) {
      setError('Invalid JSON syntax');
    }
  };

  return (
    <div>
      <Editor
        height={height}
        language="json"
        value={jsonString}
        onChange={handleChange}
        options={{
          minimap: { enabled: false },
          lineNumbers: 'on',
          scrollBeyondLastLine: false,
        }}
      />
      {error ? (
        <p className="text-red-500 text-sm mt-1">âœ— {error}</p>
      ) : (
        <p className="text-green-500 text-sm mt-1">âœ“ Valid JSON</p>
      )}
    </div>
  );
}
